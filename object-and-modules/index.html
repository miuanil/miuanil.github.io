const Employee = (function () {
 let name;
 let age;
 let salary;

 const getName = function () {
  return name;
 };

 const getAge = function () {
  return age;
 };

 const getSalary = function () {
  return salary;
 };

 return {
  setName: function (nameParam) {
   name = nameParam;
  },
  setAge: function (ageParam) {
   age = ageParam;
  },
  setSalary: function (salaryParam) {
   salary = salaryParam;
  },
  increaseSalary: function (percentage) {
   salary += (getSalary() * percentage) / 100;
  },
  incrementAge: function () {
   age = getAge() + 1;
  }
 };
})();
